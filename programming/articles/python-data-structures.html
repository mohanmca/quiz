5. Deque & BFS

from collections import deque

def bfs(adj, s):
    q = deque([s]); seen = {s}
    order = []
    while q:
        u = q.popleft(); order.append(u)
        for v in adj.get(u, []):
            if v not in seen:
                seen.add(v); q.append(v)
    return order
