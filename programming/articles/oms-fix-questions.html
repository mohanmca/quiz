3. Python: Minimal FIX Parsing

SOH = '\u0001'
def parse_fix(msg: str):
    fields = {}
    for kv in msg.split(SOH):
        if '=' in kv:
            tag, val = kv.split('=', 1)
            fields[int(tag)] = val
    return fields

ex = '8=FIX.4.4\u00019=112\u000135=D\u000149=BUY\u000156=EX\u000111=OID\u000155=BTCUSD\u0001'
print(parse_fix(ex))
