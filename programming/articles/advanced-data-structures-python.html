1. Trie

class Trie:
    def __init__(self): self.t = {}
    def insert(self, w):
        t = self.t
        for ch in w: t = t.setdefault(ch, {})
        t['#'] = True
    def search(self, w):
        t = self.t
        for ch in w:
            if ch not in t: return False
            t = t[ch]
        return '#' in t

2. Monotonic Stack

def next_greater(nums):
    res = [-1]*len(nums)
    st = []  # decreasing values' indices
    for i, x in enumerate(nums):
        while st and nums[st[-1]] < x:
            res[st.pop()] = x
        st.append(i)
    return res
