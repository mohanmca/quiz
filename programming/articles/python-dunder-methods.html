1. __repr__ / __str__

class Point:
    def __init__(self,x,y): self.x=x; self.y=y
    def __repr__(self): return f"Point(x={self.x}, y={self.y})"
    def __str__(self): return f"({self.x}, {self.y})"

2. Iteration Protocol

class Countdown:
    def __init__(self,n): self.n=n
    def __iter__(self): return self
    def __next__(self):
        if self.n<=0: raise StopIteration
        self.n -= 1; return self.n

for i in Countdown(3):
    print(i)

3. Rich Comparison

class Version:
    def __init__(self, major, minor): self.major, self.minor = major, minor
    def __lt__(self, other): return (self.major, self.minor) < (other.major, other.minor)
    def __eq__(self, other): return (self.major, self.minor) == (other.major, other.minor)

4. Context Managers

class open_write:
    def __init__(self, path): self.path = path
    def __enter__(self): self.f = open(self.path,'w'); return self.f
    def __exit__(self, exc_type, exc, tb): self.f.close(); return False

with open_write('out.txt') as f:
    f.write('hello')
