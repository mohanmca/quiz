1. Mutating Ops

a = [3,1,4]
a.append(2)           # [3,1,4,2]  (returns None)
a.extend([9,9])       # [3,1,4,2,9,9]
a.insert(1, 7)        # [3,7,1,4,2,9,9]
last = a.pop()        # removes last; returns 9
a.remove(7)           # removes first 7; ValueError if not present
a.clear()             # []

2. Copying & Aliasing

import copy
a = [[0]*3 for _ in range(3)]
b = a[:]           # shallow copy
c = copy.deepcopy(a)
a[0][0] = 99
print(b[0][0])     # 99 (shares inner lists)
print(c[0][0])     # 0  (deep copied)

# 2D pitfall
bad = [[0]*3]*3    # rows alias the same inner list
bad[0][0] = 1
print(bad)         # [[1,0,0],[1,0,0],[1,0,0]]

3. Slicing & Assignment

a = [0,1,2,3,4,5]
print(a[1:5])     # [1,2,3,4]
print(a[::2])     # [0,2,4]
# Out-of-range bounds clamp
print(a[-999:999])

# Slice assignment can resize
a[2:4] = [9,9,9]  # [0,1,9,9,9,4,5]
del a[::3]        # delete every 3rd element

4. Sorting & Stability

names = ['bob','alice','eve','al']
names.sort()                 # in place, returns None (stable)
by_len = sorted(names, key=len)           # new list
both   = sorted(names, key=lambda s:(len(s), s))

# Reverse
names.sort(reverse=True)

5. Comprehensions & Patterns

# Flatten
A = [[1,2],[3],[4,5,6]]
flat = [x for row in A for x in row]

# Frequency of values
from collections import Counter
freq = Counter([1,2,2,3,3,3])

# Unique while preserving order
uniq = list(dict.fromkeys([3,1,3,2,1]))  # [3,1,2]

6. Iterators & Reverse

a = [1,2,3]
it = reversed(a)       # iterator view
print(list(it))        # [3,2,1]

for i, x in enumerate(a, start=1):
    print(i, x)

7. Common Utilities

a = [10, -3, 5]
print(min(a, key=abs))        # -3
print(' '.join(map(str, a)))  # '10 -3 5'

# Safe concatenation vs in-place extension
b = a + [7]     # new list
a += [7]        # mutate in place (like extend)
