3. Python Consumer

from kafka import KafkaConsumer
import json
consumer = KafkaConsumer('events', bootstrap_servers=['localhost:9092'],
                         value_deserializer=lambda v: json.loads(v))
for msg in consumer:
    print(msg.value)
    consumer.commit()
