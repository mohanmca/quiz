1. P&L

def realized_pnl(entry_px, exit_px, qty, side):
    # side: +1 long, -1 short
    return (exit_px - entry_px) * qty * side

2. Funding

def funding_payment(notional, rate, side):
    # rate: signed funding rate for the period, side: +1 long, -1 short
    return notional * rate * side

3. Margin

def maintenance_margin(notional, mmr):
    return notional * mmr

def liquidation_price(entry_px, qty, balance, imr, mmr, side):
    # Simplified illustration; real exchanges apply haircuts/fees
    notional = abs(entry_px * qty)
    equity = balance + realized_pnl(entry_px, entry_px, qty, side)
    return entry_px * (1 - (equity - maintenance_margin(notional, mmr)) / notional)
