3. Python: Render Small HCL Snippet

def hcl_block(name, **kwargs):
    lines = [f"{name} {{"]
    for k,v in kwargs.items():
        v = f'"{v}"' if isinstance(v,str) else v
        lines.append(f"  {k} = {v}")
    lines.append("}")
    return "\n".join(lines)

print(hcl_block('variable', name='region', default='us-east-1'))
