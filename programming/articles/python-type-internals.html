1. Object Size

import sys
print(sys.getsizeof(0))        # header + payload
print(sys.getsizeof(10**20))   # big ints store multiple limbs
print(sys.getsizeof([]), sys.getsizeof([1,2,3]))

3. Buffer / memoryview

import array
buf = array.array('I', [1,2,3,4])
view = memoryview(buf)
view[0] = 99
print(buf.tolist())  # [99, 2, 3, 4]
